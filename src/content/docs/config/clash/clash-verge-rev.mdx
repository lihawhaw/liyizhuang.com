```javascript
const newRules = ['RULE-SET,proxy,🔰国外流量', 'RULE-SET,direct,🚀直接连接']

function main(config, profileName) {
  // return config;
  let oldRules = config.rules

  // 将新的规则添加到现有的规则中
  config.rules = newRules.concat(oldRules)

  // 返回修改后的配置
  return config
}
```
